\subsection{Before Starting Engine}

\begin{enumerate*}
  \setlength{\itemsep}{0em}

		\item Preflight Inspection - \textbf{COMPLETE}	
		\item Seats, Seatbels \& Shoulder Harness - \textbf{ADJUST and LOCK}
		\item Fuel Selector Valve - \textbf{ON FULLEST TANK}
		\item Avionics Power - \textbf{OFF}
		\item Electrical Equipment - \textbf{OFF}
		\item Brakes - \textbf{TEST and SET}
		\item Flaps - \textbf{RETRACTED}
		\item Circuit Breakers - \textbf{CHECK}

	\end{enumerate*}




\subsection{Starting Engine - Normal}

	\begin{enumerate*}
	
		\item Carburetor Heat - \textbf{OFF}
		\item Mixture - \textbf{FULL RICH}
		\item Throttle - \textbf{OPEN $\frac{1}{2}$"}
		\item Prime - \textbf{AS REQUIRED.  Skip when engine is hot.}
		\item Master - \textbf{ON}
		\item Rotating Beacon - \textbf{ON}
		\item Fuel Pump - \textbf{ON}
		\item Starter - \textbf{ENGAGE}
		\item Throttle - \textbf{1000 RPM WARM UP}
		\item Oil Pressure - \textbf{CHECK}
		\item Fuel Pump - \textbf{OFF}
		
	\end{enumerate*}



	
\subsection{After Starting}

	\begin{enumerate*}
	
		\item Lights - \textbf{AS REQUIRED}
		\item Radios - \textbf{SET}
		\item Transponder - \textbf{STANDBY, 1200}
		\item Directional Gyro - \textbf{SET}
		\item Mixture - \textbf{LEAN FOR TAXI}
		\item Parking Brake - \textbf{RELEASE}
		\item Brakes \& Steering - \textbf{CHECK}
				
	\end{enumerate*}
	
	
	
	
\subsection{Before Takeoff - Run-Up}

	\begin{enumerate*}
	
		\item Cabin Doors and Windows - \textbf{CLOSED \& LOCKED}
		\item Parking Brake - \textbf{SET}
		\item Flight Controls - \textbf{FREE \& CORRECT}
		\item Flight Instruments - \textbf{SET}
		\item Fuel Selector Valve - \textbf{ON - FULLEST TANK}
		\item Elevator Trim - \textbf{TAKEOFF}
		\item Mixture - \textbf{RICH or AS REQUIRED}
		\item Throttle - \textbf{2000 RPM}
			\begin{enumerate*}
				\item Magnetos - \textbf{CHECK}
				\item Carburetor Heat - \textbf{CHECK}
				\item Engine Instruments \& Ammeter - \textbf{CHECK}
				\item Suction Gauge - \textbf{CHECK: 5 InHg$ \pm $0.1}
			\end{enumerate*}
		\item Throttle - \textbf{IDLE, THEN 1000 RPM}
		\item Radios - \textbf{SET}
		\item Transponder - \textbf{SET, THEN ALTITUDE}
		\item Throttle Friction Lock - \textbf{ADJUST}
		\item Fuel Pump - \textbf{ON}
		\item Lights - \textbf{AS REQUIRED}
		\item Parking Brake - \textbf{RELEASE}
	
	\end{enumerate*}

\section{EMERGENCY PROCEDURES}	

\subsection{Engine Failure During Takeoff}

	\begin{enumerate*}
		
		\item Airspeed - \textbf{BEST GLIDE: 80 MPH}
		\item Fuel Selector Valve - \textbf{OFF}
		\item Fuel Pump - \textbf{OFF}
		\item Mixture - \textbf{IDLE CUTOFF}
		
	\end{enumerate*}
	
	
	
	
\subsection{Engine Failure During Flight}

	\begin{enumerate*}
	
		\item Airspeed - \textbf{BEST GLIDE - 80 MPH}
		\item Fuel Selector - \textbf{SWITCH TANKS}
		\item Magnetos - \textbf{BOTH}
		\item Fuel Pump - \textbf{ON}
		\item Carburetor Heat - \textbf{ON}
		\item Mixture - \textbf{FULL RICH}
		\item Thrittle - {FULL THROTTLE}
		\item Engine Gauges - \textbf{CHECK}
	
	\end{enumerate*}



\subsection{Forced Landings Without Engine Power}

	\begin{enumerate*}
	
		\item Airspeed - \textbf{BEST GLIDE - 80 MPH}
		\item Mixture - \textbf{IDLE CUTOFF}
		\item Fuel Selector - \textbf{OFF}
		\item Magnetos - \textbf{OFF}
		\item Flaps - \textbf{AS REQUIRED}
		\item Doors - \textbf{OPEN BEFORE TOUCHDOWN}
		\item Master Switch - \textbf{OFF}
		\item Touchdown - \textbf{TAIL LOW}
		\item Brakes - \textbf{HARD BRAKING}
	
	\end{enumerate*}
	
	
	
	
\subsection{Precautionary Landings With Power}

	\begin{enumerate*}
	
		\item Airspeed - \textbf{BEST GLIDE - 80 MPH}
		\item Flaps - \textbf{25$^\circ$}
		\item Selecting a Field - \textbf{FLY OVER TO NOTE TERRAIN AND OBSTRUCTIONS}
		\item Electrical - \textbf{OFF}
		\item Flaps - \textbf{40$^\circ$ - ON FINAL}
		\item Airspeed - \textbf{76 MPH}
	
	\end{enumerate*}
