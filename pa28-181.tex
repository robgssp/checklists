\subsection{Before Start}
\begin{checklist}

\item Belts on
\item Brakes set
\item Check circuit breakers
\item Alternate static source off
\item Carb heat cold
\item Avionics off
\item Fuel selector set

\end{checklist}


\subsection{Starting Engine}
\begin{checklist}

\item Throttle 1/4''
\item Bat, Alt on
\item Lights on
\item Left mag on
\item Fuel pump on
\item Mix rich
\item Prop clear
\item Start
\item Right mag on
\item Oil pressure check

\end{checklist}


\subsection{Before Taxi}
\begin{checklist}

\item Lean out
\item ATIS
\item Brake check
\item Gyros check

\end{checklist}


\subsection{Run-Up}
\begin{checklist}

\item Brakes set
\item Flight controls free \& correct
\item Switch tanks
\item Throttle to 2000rpm
\item Mag check - 175 max, 50 diff
\item Carb heat check
\item Fuel pump off, check, on
\item Vacuum 4.8 to 5.2 in. Hg.
\item Oil temp, pressure
\item AC check
\item Ammeter check
\item Annunciators check, test
\item Throttle idle, 1000rpm

\end{checklist}


\subsection{Before Takeoff}
\begin{checklist}

\item Flight instruments check, set
\item Lights set
\item Fuel selector correct
\item Fuel pump on
\item Seats, belts set
\item Flaps set
\item Trim set
\item AC off

\end{checklist}

\subsection{Speeds}

\begin{itemize*}
\item $V_{x}$: 64kts
\item $V_y$: 76kts
\item $V_{glide}$: 76kts
\item $V_a$: 89 to 113kts
\item final: 66kts
\item Cruise climb: 87kts
\end{itemize*}


\subsection{Cruise}
\begin{checklist}

\item Fuel pump off
\item Lean out
\item Lights set
\item Switch tanks every 15min, fuel pump when switching

\end{checklist}


\subsection{Landing}
\begin{checklist}

\item Fuel selector to fullest
\item Belts
\item Fuel pump on
\item Carb heat as req'd

\end{checklist}


\subsection{After Landing}
\begin{checklist}

\item Flaps up
\item Lean out
\item Lights

\end{checklist}


\subsection{Shutdown}
\begin{checklist}

\item Flaps up
\item Fuel pump off
\item Radios off
\item Mix to cut-off
\item Mags off
\item Master, electricals off
\item Control lock on

\end{checklist}


\section{EMERGENCY PROCEDURES}

\subsection{Engine Fire On Start}
\begin{checklist}
\item Keep cranking
\item Mix to cut-off
\item Throttle open
\item Fuel pump off
\item Fuel selector off
\end{checklist}

\subsection{Engine Failure On Takeoff}
\begin{checklist}

\item Fuel selector switch
\item Fuel pump on
\item Mix rich
\item Carb heat on

\end{checklist}


\subsection{Engine Failure In Flight}
\begin{checklist}

\item Airspeed $\geq$76kias
\item Fuel selector switch
\item Fuel pump on
\item Mix rich
\item Carb heat on
\item Check gauges for issue

  If power is restored:

\item Carb heat off
\item Fuel pump off
\end{checklist}


\subsection{Power Off Landing}
\begin{checklist}

\item Flaps as desired
\item Throttle closed
\item Mix cutoff
\item Mags off
\item Master off
\item Fuel selector off
\item Seatbelts tight

\end{checklist}


\subsection{Electrical Fire In Flight}
\begin{checklist}

\item Batt, Alternator off
\item Vents open
\item Cabin heat off
\item Land as soon as possible

\end{checklist}


\subsection{Engine Fire In Flight}
\begin{checklist}

\item Fuel selector off
\item Throttle closed
\item Mix cutoff
\item Fuel pump off
\item Heater, defroster off
\item Make power-off landing

\end{checklist}


\subsection{Electrical Failures}
\begin{checklist}

\item If alt annunciator, check ammeter
\item If ammeter is zero, Alt off
\item Reduce electrical load
\item Check Alt circuit breaker
\item Alt switch on
\item Alt switch off if not fixed

\end{checklist}

\subsection{Electrical Overload (+20A On Ammeter)}
\begin{checklist}

\item Alt switch on
\item Bat switch off

  if load is reduced:

\item reduce electrical load

  if load isn't reduced:

\item Alt switch off
\item Bat switch as req'd
\item Land as soon as possible

\end{checklist}
