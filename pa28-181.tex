\subsection{Before Start}
\begin{checklist}

\item Belts on
\item Brakes set
\item Check circuit breakers
\item Alternate static source off
\item Carb heat cold
\item Avionics off
\item Fuel selector set

\end{checklist}


\subsection{Starting Engine - Normal}
\begin{checklist}
  
\item Throttle 1/4''
\item Bat, Alt on
\item Lights on
\item Left mag on
\item Fuel pump on
\item Mix rich
\item Prop clear
\item Start
\item Right mag on
\item Oil pressure check
  
\end{checklist}


\subsection{Before Taxi}
\begin{checklist}

\item Lean out
\item ATIS
\item Brake check
\item Gyros check

\end{checklist}


\subsection{Run-Up}
\begin{checklist}

\item Brakes set
\item Flight controls free \& correct
\item Switch tank
\item Throttle to 2000rpm
\item Mag check - 175 max, 50 diff
\item Carb heat check
\item Fuel pump off, check, on
\item Vacuum 4.8 to 5.2 in. Hg.
\item Oil temp, pressure
\item AC check
\item Ammeter check
\item Annunciators check, test
\item Throttle idle, 1000rpm

\end{checklist}


\subsection{Before Takeoff}
\begin{checklist}
  
\item Flight instruments check, set
\item Lights set
\item Fuel selector correct
\item Fuel pump on
\item Seats, belts set
\item Flaps set
\item Trim set
\item AC off
  
\end{checklist}

\subsection{Speeds}

\begin{itemize*}
\item $V_{x}$: 64kts
\item $V_y$: 76kts
\item $V_{glide}$: 76kts
\item $V_a$: 89 to 113kts
\item final: 66kts
\item Cruise climb: 87kts
\end{itemize*}

	
\subsection{Cruise}
\begin{checklist}
  
\item Fuel pump off
\item Lean out
\item Lights set
\item Switch tanks every 15min, fuel pump when switching
  
\end{checklist}


\subsection{Landing}
\begin{checklist}
  
\item Fuel selector to fullest
\item Belts
\item Fuel pump on
\item Carb heat as req'd

\end{checklist}
	
	
\subsection{After Landing}
\begin{checklist}
	
\item Flaps up
\item Lean out
\item Lights
	
\end{checklist}
	
	
\subsection{Shutdown}
\begin{checklist}

\item Flaps up
\item Fuel pump off
\item Radios off
\item Mix to cut-off
\item Mags off
\item Master, electricals off
\item Control lock on
  
\end{checklist}
	
	
\section{EMERGENCY PROCEDURES}

\subsection{Engine fire on start}
\begin{checklist}
\item Keep cranking
\item Mix to cut-off
\item Throttle open
\item Fuel pump off
\item Fuel selector off
\end{checklist}
\subsection{Engine Failure During Takeoff}

\begin{checklist}
  
\item Airspeed - \textbf{BEST GLIDE: 80 MPH}
\item Fuel Selector Valve - \textbf{OFF}
\item Fuel Pump - \textbf{OFF}
\item Mixture - \textbf{IDLE CUTOFF}
  
\end{checklist}




\subsection{Engine Failure During Flight}

\begin{checklist}
  
\item Airspeed - \textbf{BEST GLIDE - 80 MPH}
\item Fuel Selector - \textbf{SWITCH TANKS}
\item Magnetos - \textbf{BOTH}
\item Fuel Pump - \textbf{ON}
\item Carburetor Heat - \textbf{ON}
\item Mixture - \textbf{FULL RICH}
\item Thrittle - {FULL THROTTLE}
\item Engine Gauges - \textbf{CHECK}
  
\end{checklist}



\subsection{Forced Landings Without Engine Power}

\begin{checklist}
  
\item Airspeed - \textbf{BEST GLIDE - 80 MPH}
\item Mixture - \textbf{IDLE CUTOFF}
\item Fuel Selector - \textbf{OFF}
\item Magnetos - \textbf{OFF}
\item Flaps - \textbf{AS REQUIRED}
\item Doors - \textbf{OPEN BEFORE TOUCHDOWN}
\item Master Switch - \textbf{OFF}
\item Touchdown - \textbf{TAIL LOW}
\item Brakes - \textbf{HARD BRAKING}
  
\end{checklist}




\subsection{Precautionary Landings With Power}

\begin{checklist}
  
\item Airspeed - \textbf{BEST GLIDE - 80 MPH}
\item Flaps - \textbf{25$^\circ$}
\item Selecting a Field - \textbf{FLY OVER TO NOTE TERRAIN AND OBSTRUCTIONS}
\item Electrical - \textbf{OFF}
\item Flaps - \textbf{40$^\circ$ - ON FINAL}
\item Airspeed - \textbf{76 MPH}
  
\end{checklist}
